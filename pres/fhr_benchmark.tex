        \begin{column}{0.5\textwidth}
            \textbf{Key Takeaways} 
            \begin{itemize}
                \visible<2->{\item Peak in Group 1 fast neutrons born in assembly 
                diamond's center} 
                \visible<3->{\item Fast neutrons are moderated in graphite matrix and 
                structure} 
                \visible<4->{\item Outer diamond's sides absorb the moderated thermal 
                neutrons and \textbf{geometrically shield the assembly diamond's center 
                from neutron thermal flux}}
                \visible<5->{\item Results in a dip in thermal Group 3 flux in the assembly 
                diamond's center}
                \item This \textbf{self-shielding effect is more pronounced in Case 6A} 
            \end{itemize}
        \end{column}
    \end{columns}
